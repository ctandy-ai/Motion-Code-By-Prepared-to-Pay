from datetime import datetime
from typing import Dict, Any, List, Optional
from fastapi import FastAPI, HTTPException, Body

# ---------- tiny file store ----------
def _path(fname: str) -> str:
    return os.path.join(DATA_DIR, fname)

def read_json(fname: str, default: Any):
    p = _path(fname)
    if not os.path.exists(p):
        write_json(fname, default)
        return default
    with open(p, "r", encoding="utf-8") as f:
        return json.load(f)

def write_json(fname: str, obj: Any):
    tmp = _path(fname + ".tmp")
    with open(tmp, "w", encoding="utf-8") as f:
        json.dump(obj, f, indent=2)
    shutil.move(tmp, _path(fname))

    lib = get_library()
    idx = next((i for i, e in enumerate(lib) if e["id"] == item["id"]), -1)
    if idx >= 0:
        lib[idx] = item
    else:
        lib.append(item)
    save_library(lib)

@app.post("/api/library/override")
def api_library_override(payload: Dict[str, Any] = Body(...)):
    """
    payload = { "exercise_id": "...", "variables_default": {...}, "belt_min":"White","belt_max":"Blue" }
    """
    ex_id = payload.get("exercise_id")
    ex = find_exercise(ex_id)
    if not ex:
        raise HTTPException(404, f"exercise {ex_id} not found")
    if "variables_default" in payload:
        ex["variables_default"] = json.dumps(payload["variables_default"])
    if "belt_min" in payload: ex["belt_min"] = payload["belt_min"]
    if "belt_max" in payload: ex["belt_max"] = payload["belt_max"]
    api_library_upsert(ex)
    return {"ok": True}

@app.post("/api/templates/save")
def api_templates_save(payload: Dict[str, Any] = Body(...)):
    """
    payload = { "name": "Template Name", "belt":"Blue", "blocks":[{slot, exercise_id}] }
    """
    name = payload.get("name")
    if not name:
        raise HTTPException(400, "name required")
    tpl = get_templates()
    tpl[name] = {"belt": payload.get("belt","White"), "blocks": payload.get("blocks", [])}
    save_templates(tpl)
    return {"ok": True}

@app.post("/api/templates/duplicate")
def api_templates_duplicate(payload: Dict[str, Any] = Body(...)):
    """
    payload = {"from":"Old Name","to":"New Name"}
    """
    src = payload.get("from"); dst = payload.get("to")
    tpl = get_templates()
    if src not in tpl:
        raise HTTPException(404, "source template not found")
    if dst in tpl:
        raise HTTPException(400, "target already exists")
    tpl[dst] = json.loads(json.dumps(tpl[src]))  # deep copy
    save_templates(tpl)
    return {"ok": True}

@app.get("/api/program/12m")
def api_program():
    return get_program()

@app.post("/api/program/apply")
def api_program_apply(payload: Dict[str, Any] = Body(...)):
    """
    payload = {
      "week_idx": 0, "day_idx": 0,
      "blocks": [{"slot":"A1","exercise_id":"BARBELL_BACK_SQUAT"}, ...]
    }
    """
    prog = get_program()
    w = payload["week_idx"]; d = payload["day_idx"]
    if w < 0 or w >= len(prog["weeks"]): raise HTTPException(400, "bad week_idx")
    if d < 0 or d >= 7: raise HTTPException(400, "bad day_idx")
    prog["weeks"][w]["days"][d]["blocks"] = payload.get("blocks", [])
    save_program(prog)
    return {"ok": True}

@app.post("/api/program/apply_template")
def api_program_apply_template(payload: Dict[str, Any] = Body(...)):
    """
    payload = { "template_name":"...", "week_idx": 0, "day_idx": 0 }
    """
    tpl = get_templates()
    name = payload["template_name"]
    if name not in tpl: raise HTTPException(404, "template not found")
    blocks = tpl[name]["blocks"]
    return api_program_apply({"week_idx": payload["week_idx"], "day_idx": payload["day_idx"], "blocks": blocks})

# ---------- Athlete "today" feed ----------
@app.get("/api/athlete/today")
def api_athlete_today(week_idx: int = 0, day_idx: int = 0):
    prog = get_program()
    try:
        day = prog["weeks"][week_idx]["days"][day_idx]
    except Exception:
        raise HTTPException(400, "invalid week/day")
    # expand blocks with exercise metadata
    lib_index = {e["id"]: e for e in get_library()}
    expanded = []
    for b in day.get("blocks", []):
        ex = lib_index.get(b["exercise_id"]); 
        if ex:
            v = json.loads(ex["variables_default"]) if ex.get("variables_default") else {}
            expanded.append({
                "slot": b.get("slot",""),
                "exercise": ex["name"],
                "sets": v.get("sets"), "reps": v.get("reps"), "intensity": v.get("intensity")
            })
    return {"week_idx":week_idx, "day_idx":day_idx, "blocks": expanded}
